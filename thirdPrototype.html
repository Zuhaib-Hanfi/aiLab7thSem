from collections import deque

def bfs(graph, start):
    """
    Perform Breadth First Search on an unweighted graph.

    :param graph: dict with node -> list of neighbors
    :param start: starting node for BFS
    :return: list of nodes in BFS traversal order
    """
    visited = set()
    queue = deque()
    order = []

    visited.add(start)
    queue.append(start)

    while queue:
        node = queue.popleft()
        order.append(node)

        for neighbor in graph.get(node, []):
            if neighbor not in visited:
                visited.add(neighbor)
                queue.append(neighbor)

    return order


if __name__ == "__main__":

        "B": ["D", "E"],
        "C": ["F"],
        "D": [],
        "E": ["F"],
        "F": []
    }

    start_node = "A"
    traversal = bfs(graph, start_node)
    print("BFS Traversal from node", start_node, ":", traversal)

def dfs_recursive(graph, node, visited, order):
    visited.add(node)
    order.append(node)

    for neighbor in graph.get(node, []):
        if neighbor not in visited:
            dfs_recursive(graph, neighbor, visited, order)

    :return: list of nodes in DFS traversal order
    """
    visited = set()
    order = []
    dfs_recursive(graph, start, visited, order)

def print_board(board):
    print("\nCurrent board:")
    for i in range(3):
        row = " | ".join(board[i * 3:(i + 1) * 3])
        print(" " + row)
        if i < 2:
            print("---+---+---")


def check_winner(board, mark):
    # Rows
    for i in range(0, 9, 3):
        if board[i] == board[i + 1] == board[i + 2] == mark:
            return True

    # Columns
    for i in range(3):
        if board[i] == board[i + 3] == board[i + 6] == mark:
            return True

    # Diagonals
    if board[0] == board[4] == board[8] == mark:
        return True
    if board[2] == board[4] == board[6] == mark:
        return True

    return False


def tic_tac_toe():
    board = [" "] * 9
    current_player = "X"

    print("Welcome to Tic-Tac-Toe!")
    print("Players: X and O")
    print("Enter position (1-9) as shown below:\n")
    print(" 1 | 2 | 3")
    print("---+---+---")
    print(" 4 | 5 | 6")
    print("---+---+---")
    print(" 7 | 8 | 9")

    for move in range(9):
        print_board(board)
        try:
            pos = int(input(f"\nPlayer {current_player}, enter your move (1-9): "))
        except ValueError:
            print("Invalid input. Please enter a number from 1 to 9.")
            continue

        if pos < 1 or pos > 9:
            print("Position out of range. Try again.")
            continue

        if board[pos - 1] != " ":
            print("Position already taken. Try again.")
            continue

        board[pos - 1] = current_player

        if check_winner(board, current_player):
            print_board(board)
            print(f"\nPlayer {current_player} wins! ðŸŽ‰")
            return

        # Switch player
        current_player = "O" if current_player == "X" else "X"

    print_board(board)
    print("\nIt's a draw!")


if __name__ == "__main__":
    tic_tac_toe()

from collections import deque

def get_neighbors(state):
    neighbors = []
    idx = state.index("_")
    row, col = divmod(idx, 3)

    moves = {
        "U": (row - 1, col),
        "D": (row + 1, col),
        "L": (row, col - 1),
        "R": (row, col + 1),
    }

    for move, (r, c) in moves.items():
        if 0 <= r < 3 and 0 <= c < 3:
            new_idx = r * 3 + c
            state_list = list(state)
            state_list[idx], state_list[new_idx] = state_list[new_idx], state_list[idx]
            neighbors.append((move, "".join(state_list)))
    return neighbors

def is_solvable(state):
    tiles = [c for c in state if c != "_"]
    inversions = 0
    for i in range(len(tiles)):
        for j in range(i+1, len(tiles)):
            if int(tiles[i]) > int(tiles[j]):
                inversions += 1
    return inversions % 2 == 0

def bfs_8_puzzle(start, goal):
    if start == goal:
        return []
    if not is_solvable(start):
        return None  # unsolvable

    queue = deque([start])
    parent = {start: None}
    action = {start: None}

    while queue:
        state = queue.popleft()
        if state == goal:
            # reconstruct path as list of (move, state) excluding the initial state's None move
            path = []
            while parent[state] is not None:
                path.append((action[state], state))
                state = parent[state]
            path.reverse()
            return path

        for move, neighbor in get_neighbors(state):
            if neighbor not in parent:
                parent[neighbor] = state
                action[neighbor] = move
                queue.append(neighbor)
    return None

def print_state(state):
    for i in range(0, 9, 3):
        print(" ".join(state[i:i + 3]))
    print()

if __name__ == "__main__":
    # Example 1: user's original start (UNSOLVABLE)
    start_state = "1253_4678"

from collections import deque

# Example: Jug capacities 4 and 3 liters, target 2 liters.


def get_next_states(state, cap_x, cap_y):
    x, y = state
    states = []

    # Fill jug X
    states.append((cap_x, y))
    # Fill jug Y
    states.append((x, cap_y))

    # Empty jug X
    states.append((0, y))
    # Empty jug Y
    states.append((x, 0))

    # Pour X -> Y
    transfer = min(x, cap_y - y)
    states.append((x - transfer, y + transfer))

    # Pour Y -> X
    transfer = min(y, cap_x - x)
    states.append((x + transfer, y - transfer))

    return states


def bfs_water_jug(cap_x, cap_y, target):
    start = (0, 0)
    queue = deque([start])
    parent = {start: None}

    while queue:
        state = queue.popleft()
        x, y = state

        if x == target or y == target:
            # Reconstruct path
            path = []
            while state is not None:
                path.append(state)
                state = parent[state]
            path.reverse()
            return path

        for next_state in get_next_states(state, cap_x, cap_y):
            if next_state not in parent:
                parent[next_state] = state
                queue.append(next_state)

    return None


if __name__ == "__main__":

    cap_y = 3
    target = 2

    solution_path = bfs_water_jug(cap_x, cap_y, target)

    if solution_path is None:
        print("No solution found.")
    else:
        print("Solution steps (X, Y):")
        for step, state in enumerate(solution_path):
            print(f"Step {step}: {state}")

import itertools

def travelling_salesman(dist_matrix, start=0):
    """
    Brute force solution for TSP.
    dist_matrix: 2D list where dist_matrix[i][j] is distance from i to j.
    start: starting city index.
    """
    n = len(dist_matrix)
    cities = list(range(n))

    # Generate all permutations of cities except the starting one
    other_cities = [c for c in cities if c != start]

    best_path = None
    best_cost = float("inf")

    for perm in itertools.permutations(other_cities):
        path = [start] + list(perm) + [start]
        cost = 0

        for i in range(len(path) - 1):
            cost += dist_matrix[path[i]][path[i + 1]]

        if cost < best_cost:
            best_cost = cost
            best_path = path

    return best_path, best_cost


if __name__ == "__main__":
    # Example with 4 cities
    dist_matrix = [
        [0, 10, 15, 20],
        [10, 0, 35, 25],
        [15, 35, 0, 30],
        [20, 25, 30, 0],
    ]

    path, cost = travelling_salesman(dist_matrix, start=0)
    print("Best tour:", path)
    print("Minimum cost:", cost)

def tower_of_hanoi(n, source, auxiliary, target):
    """
    Move n disks from 'source' to 'target' using 'auxiliary' as helper.
    """
    if n == 1:
        print(f"Move disk 1 from {source} to {target}")
        return

    tower_of_hanoi(n - 1, source, target, auxiliary)
    print(f"Move disk {n} from {source} to {target}")
    tower_of_hanoi(n - 1, auxiliary, source, target)

# Simple version of Monkey and Banana problem.
# States are represented as:
# (monkey_position, box_position, monkey_on_box, banana_grabbed)
# positions: "door", "middle", "window"
# monkey_on_box: True / False
# banana_grabbed: True / False

from collections import deque

positions = ["door", "middle", "window"]


def get_next_states(state):
    monkey_pos, box_pos, on_box, has_banana = state
    states = []

    if has_banana:
        return states  # goal reached; no further actions

    # 1. Monkey walks to another position (without box)
    for p in positions:
        if p != monkey_pos and not on_box:
            states.append(("walk to " + p, (p, box_pos, on_box, has_banana)))

    # 2. Monkey pushes the box if at same position and not on box
    if monkey_pos == box_pos and not on_box:
        for p in positions:
            if p != monkey_pos:
                states.append(
                    ("push box to " + p, (p, p, on_box, has_banana))
                )

    # 3. Monkey climbs onto the box
    if monkey_pos == box_pos and not on_box:
        states.append(("climb on box", (monkey_pos, box_pos, True, has_banana)))

    # 4. Monkey climbs down from the box
    if on_box:
        states.append(("climb down", (monkey_pos, box_pos, False, has_banana)))

    # 5. Monkey grabs banana (only if under banana = middle and on box)
    if monkey_pos == "middle" and box_pos == "middle" and on_box:
        states.append(
            ("grab banana", (monkey_pos, box_pos, on_box, True))
        )

    return states


def bfs_monkey_banana():
    start = ("door", "window", False, False)
    goal_condition = lambda s: s[3] is True  # has_banana == True

    queue = deque([start])
    parent = {start: None}
    action = {start: None}

    while queue:
        state = queue.popleft()
        if goal_condition(state):
            # reconstruct path
            path = []
            while state is not None:
                path.append((action[state], state))
                state = parent[state]
            path.reverse()
            return path

        for act, next_state in get_next_states(state):
            if next_state not in parent:
                parent[next_state] = state
                action[next_state] = act
                queue.append(next_state)

    return None


if __name__ == "__main__":
    path = bfs_monkey_banana()
    if path is None:
        print("No solution found.")
    else:
        print("Solution to Monkey-Banana problem:")
        for step, (act, state) in enumerate(path):
            print(f"Step {step}:")
            if act:
                print("  Action:", act)
            print("  State :", state)

def alpha_beta(node_index, depth, alpha, beta, maximizing_player, values, branching_factor):
    if depth == 0:
        return values[node_index]

    if maximizing_player:
        max_eval = float("-inf")
        for i in range(branching_factor):
            child_index = node_index * branching_factor + i  # zero-based child index
            eval_value = alpha_beta(
                child_index,
                depth - 1,
                alpha,
                beta,
                False,
                values,
                branching_factor,
            )
            max_eval = max(max_eval, eval_value)
            alpha = max(alpha, eval_value)
            if beta <= alpha:
                break
        return max_eval
    else:
        min_eval = float("inf")
        for i in range(branching_factor):
            child_index = node_index * branching_factor + i  # zero-based child index
            eval_value = alpha_beta(
                child_index,
                depth - 1,
                alpha,
                beta,
                True,
                values,
                branching_factor,
            )
            min_eval = min(min_eval, eval_value)
            beta = min(beta, eval_value)
            if beta <= alpha:
                break
        return min_eval


if __name__ == "__main__":
    branching_factor = 2
    depth = 3  # depth from root to leaves
    leaf_values = [3, 5, 6, 9, 1, 2, 0, -1]

    best_value = alpha_beta(
        node_index=0,
        depth=depth,
        alpha=float("-inf"),
        beta=float("inf"),
        maximizing_player=True,
        values=leaf_values,
        branching_factor=branching_factor,
    )

    print("Best achievable value for maximizing player:", best_value)

# 8-Queens problem solved using backtracking.

N = 8


def is_safe(board, row, col):
    # Check column
    for i in range(row):
        if board[i] == col:
            return False

    # Check upper-left diagonal
    i, j = row - 1, col - 1
    while i >= 0 and j >= 0:
        if board[i] == j:
            return False
        i -= 1
        j -= 1

    # Check upper-right diagonal
    i, j = row - 1, col + 1
    while i >= 0 and j < N:
        if board[i] == j:
            return False
        i -= 1
        j += 1

    return True


def solve_queens(row, board, solutions):
    if row == N:
        solutions.append(board.copy())
        return

    for col in range(N):
        if is_safe(board, row, col):
            board[row] = col
            solve_queens(row + 1, board, solutions)
            board[row] = -1  # backtrack


def print_solution(board):
    for r in range(N):
        row_str = ""
        for c in range(N):
            row_str += "Q " if board[r] == c else ". "
        print(row_str)
    print()


if __name__ == "__main__":
    board = [-1] * N
    solutions = []
    solve_queens(0, board, solutions)

    print(f"Total solutions found: {len(solutions)}\n")
    print("Showing first 3 solutions:\n")
    for i, sol in enumerate(solutions[:3], start=1):
        print(f"Solution {i}:")
        print_solution(sol)
